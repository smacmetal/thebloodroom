import { NextResponse } from 'next/server';
import { getSignedGetUrl } from '@/lib/s3';

export async function GET(req: Request) {
  try {
    const url = new URL(req.url);
    const key = url.searchParams.get('key');
    if (!key) return NextResponse.json({ ok: false, error: 'Missing key' }, { status: 400 });

    const signed = await getSignedGetUrl(key, 300); // 5 minutes
    return NextResponse.json({ ok: true, url: signed });
  } catch (e: any) {
    console.error('[attachments:sign] ERROR', e);
    return NextResponse.json({ ok: false, error: e?.message || 'Sign failed' }, { status: 500 });
  }
}
